<script>


// ARRAY OF OBJECTS

const movies=[

{
id:1,
name:"Avengers",
price:250,
image:"https://via.placeholder.com/200"
},


{
id:2,
name:"Avatar",
price:300,
image:"https://via.placeholder.com/200"
},


{
id:3,
name:"Batman",
price:200,
image:"https://via.placeholder.com/200"
}

];




// booking array

let bookings=[];



// DOM

const movieList =
document.getElementById("movieList");


const booking =
document.getElementById("booking");


const total =
document.getElementById("total");





// ASYNC CONCEPT

async function loadMovies(){


let data = await fetchMovies();


displayMovies(data);


}




function fetchMovies(){

return new Promise(resolve=>{


setTimeout(()=>{

resolve(movies);


},1000);


});


}






// DISPLAY MOVIE CARDS

function displayMovies(data){



data.forEach(movie=>{


movieList.innerHTML += `


<div class="movie-card">


<img src="${movie.image}">


<h2>${movie.name}</h2>


<p>
Price ₹${movie.price}
</p>


<button onclick="bookTicket(${movie.id})">

Book Ticket

</button>


</div>


`;



});


}






// EVENT + FUNCTION

function bookTicket(id){



let selectedMovie =
movies.find(movie=>movie.id===id);



bookings.push(selectedMovie);



showBooking();



}





// SHOW BOOKING

function showBooking(){



booking.innerHTML="";


let amount=0;



bookings.forEach(movie=>{


amount += movie.price;



booking.innerHTML += `

<p>
🎟 ${movie.name}
-
₹${movie.price}

</p>


`;



});



total.innerHTML=amount;


}







// LOCAL STORAGE EVENT

document
.getElementById("saveBtn")
.addEventListener(
"click",
()=>{


localStorage.setItem(

"movieBooking",

JSON.stringify(bookings)

);


alert("Booking Saved Successfully");


});








// LOAD SAVED DATA


window.onload=function(){



let saved =
localStorage.getItem("movieBooking");



if(saved){


bookings =
JSON.parse(saved);


showBooking();


}



loadMovies();



}




</script>
